```cpp=

private:
    stack<T> stack1;
    stack<T> stack2;
public:
    MyQueue()
    {
        stack1 = new stack<T>();
        stack2 = new stack<T>();
    }
    // enqueue(): Add an element at the rear side of MyQueue
    void enqueue(T e)
    {
        stack1.push(e);
    }
    // dequeue(): Remove the front element from MyQueue
    T dequeue(T e)
    {
        if( (__I__) .isEmpty() )
            while( !(__II__).isEmpty() )
                (__III__).push( (__IV__) .pop() );

            temp = (__VI__).pop();
        return temp;
    }
}
```
